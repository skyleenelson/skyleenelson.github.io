bar1 <- ggplot() + 
  geom_bar(data=teamvsopp, aes(x=Team, weight=PTS, fill=Type)) +

  coord_flip() + 
  labs(
    x="",
    y="",
    title= "Total regular season points made vs points allowed",
    subtitle = "Golden State gave up more points in the regular season but still finished with a better record.",
    caption = " ") +
  theme_minimal() + 
  theme(
    plot.title = element_text(size = 12, face = "bold"),

    panel.grid.minor = element_blank(),
    )
